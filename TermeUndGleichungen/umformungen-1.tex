  \newcommand{\gap}{\;\;}%
  \ifproblem\problem
    Löse die Klammern auf, und fasse so weit wie möglich zusammen:
    \begin{align*}
      \text{a)}\gap & 3a+(2b-c)-(2a+3c-b) &
      \text{e)}\gap & 2a^2+3a-(a+5)-(1-3a^2)
      \\
      \text{b)}\gap & 2x+5y-(y-3x+2)+(x-8) &
      \text{f)}\gap & y^3-6y+(2y^2+3y-4)-(y^3-5)
      \\
      \text{c)}\gap & 3a-8b+(11a+4)-(5b-a+3) &
      \text{g)}\gap & 3x^2+y^2-(x^2-xy-y^2)+(5y^2-5xy)
      \\
      \text{d)}\gap & 9+3e-5f-(e+f-1)+(7-4e) &
      \text{h)}\gap & 3ab+6+(a^2-2ab-5)-(4b^2-a^2+1)
    \end{align*}
  \fi

  \ifoutcome\outcome
    % first line with a fix width
    \newcommand{\toprow}[1]
    {%
      \makebox[20em][l]
      {%
        \ensuremath
        {%
          \displaystyle
          \phantom{\,=\:\,}%

      }%

    % solition of exercise a)
    \newcommand{\solutionA}
    {%
      \begin{equation*}
        \begin{split}
          \text{a)} &\toprow{3a+(2b-c)-(2a+3c-b)} \\
                    &=       3a+2b-c-2a-3c+b      \\
                    &=       a+3b-4c
        \end{split}
      \end{equation*}
    }%
    % solition of exercise b)
    \newcommand{\solutionB}
    {%
      \begin{equation*}
        \begin{split}
          \text{b)} &\toprow{2x+5y-(y-3x+2)+(x-8)} \\
                    &=       2x+5y-y+3x-2+x-8      \\
                    &=       6x+4y-10
        \end{split}
      \end{equation*}
    }%
    % solition of exercise c)
    \newcommand{\solutionC}
    {%
      \begin{equation*}
        \begin{split}
          \text{c)} &\toprow{3a-8b+(11a+4)-(5b-a+3)} \\
                    &=     3a-8b+11a+4-5b+a-3        \\
                    &=     15a-13b+1
        \end{split}
      \end{equation*}
    }%
    % solition of exercise d)
    \newcommand{\solutionD}
    {%
      \begin{equation*}
        \begin{split}
          \text{d)} &\toprow{9+3e-5f-(e+f-1)+(7-4e)} \\
                    &=     9+3e-5f-e-f+1+7-4e        \\
                    &=     -2e-6f+17
        \end{split}
      \end{equation*}
    }%
    % solition of exercise e)
    \newcommand{\solutionE}
    {%
      \begin{equation*}
        \begin{split}
          \text{e)} &\toprow{2a^2+3a-(a+5)-(1-3a^2)} \\
                    &=     2a^2+3a-a-5-1+3a^2        \\
                    &=     5a^2+2a-6
        \end{split}
      \end{equation*}
    }%
    % solition of exercise f)
    \newcommand{\solutionF}
    {%
      \begin{equation*}
        \begin{split}
          \text{f)} &\toprow{y^3-6y+(2y^2+3y-4)-(y^3-5)} \\
                    &=       y^3-6y+2y^2+3y-4-y^3+5      \\
                    &=       2y^2-3y+1
        \end{split}
      \end{equation*}
    }%
    % solition of exercise g)
    \newcommand{\solutionG}
    {%
      \begin{equation*}
        \begin{split}
          \text{g)} &\toprow{3x^2+y^2-(x^2-xy-y^2)+(5y^2-5xy)} \\
                    &=       3x^2+y^2-x^2+xy+y^2+5y^2-5xy      \\
                    &=       2x^2+7y^2-4xy
        \end{split}
      \end{equation*}
    }%
    % solition of exercise h)
    \newcommand{\solutionH}
    {%
      \begin{equation*}
        \begin{split}
          \text{h)} &\toprow{3ab+6+(a^2-2ab-5)-(4b^2-a^2+1)} \\
                    &=       3ab+6+a^2-2ab-5-4b^2+a^2-1      \\
                    &=       a^2+a^2-4b^2+3ab-2ab+6-5-1      \\
                    &=       2a^2-4b^2+ab
        \end{split}
      \end{equation*}
    }%
