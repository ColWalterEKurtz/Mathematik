\begin{exercise}
      {ID-d848d92104626e00eca4d41c34ad16ae7126df90}
      {Blumenbeet}
  \ifproblem\problem\par
    Neben einem Weg befindet sich ein \simeter{1.20} breiter Wiesenstreifen. Darauf
    soll ein Blumenbeet mit einer Gesamtfläche von \simm{8.1} angelegt werden.
    Wie lang wird das Beet?
  \fi
  %\ifoutline\outline\par
  %\fi
  \ifoutcome\outcome\par
    Um die geplante Gesamtfläche von \simm{8.1} zu erreichen, muss
    das Blumenbeet eine Länge von \simeter{6.75} bekommen.
  \fi
\end{exercise}
