\begin{exercise}
      {ID-9609ef20c712d0d7f093bc0a1f6abb02f751160b}
      {Zollgebühren}
  \ifproblem\problem
    Für eine Sendung Waren im Wert von \eur{16000} wurden \eur{3200} Zoll gezahlt.
    Wie hoch sind die Zollgebühren bei einem Wert von \eur{7000}?
  \fi
  %\ifoutline\outline
  %\fi
  \ifoutcome\outcome
    Für Waren im Wert von \eur{7000} muss man Zollgebühren in Höhe von \eur{1400} zahlen.
  \fi
\end{exercise}
