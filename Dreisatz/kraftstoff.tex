\begin{exercise}
    {ID-4193fce4198e4126b52856536df0d2f4a2950fb1}
    {Kraftstoff}
\ifproblem\problem\par
  Für eine Strecke von \sikm{340} braucht ein Pkw \sil{28.56} Benzin.
  \begin{enumerate}[a)]
    \item Wie viel Kraftstoff braucht der Pkw für \sikm{650}?
    \item Wie viele Kilometer kann er mit einem Tankinhalt von \sil{60.48} zurücklegen?
  \end{enumerate}
\fi
%\ifoutline\outline\par
%\fi
\ifoutcome\outcome\par
  \begin{enumerate}[a)]
    \item Für \sikm{650} benötigt der Pkw \sil{54.6} Kraftstoff.
    \item Mit einem Tankinhalt von \sil{60.48} kann der Pkw \sikm{720} zurücklegen.
  \end{enumerate}
\fi
\end{exercise}
