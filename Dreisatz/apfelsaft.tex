      {Apfelsaft}
  \ifproblem\problem
    Für \num{80} Liter Apfelsaft braucht man \sikg{100} Äpfel.
    \begin{enumerate}[a)]
      \item Wie viele Äpfel werden für \num{200} Liter Apfelsaft benötigt?
      \item Wie viel Apfelsaft presst man aus \sikg{120} Äpfeln?
    \end{enumerate}
  \fi
  %\ifoutline\outline
  %\fi
  \ifoutcome\outcome
    \begin{enumerate}[a)]
      \item Für \num{200} Liter Apfelsaft benötigt man \sikg{250} Äpfel.
      \item Aus \sikg{120} Äpfeln presst man \num{96} Liter Apfelsaft.
    \end{enumerate}
  \fi
\end{exercise}
