\begin{exercise}
      {ID-5ba8fe30e46c16f2271971e17850e7383f598293}
      {Fett}
  \ifproblem\problem
    Butter hat einen Fettgehalt von \pc{82}, Crème fraîche enthält \pc{30}
    Fett. Wie viel Gramm Butter enthält dieselbe Menge Fett wie ein Becher
    mit \sig{125} Crème fraîche?
  \fi
  %\ifoutline\outline
  %\fi
  \ifoutcome\outcome
    \begin{equation*}
      0,\!82\cdot x=0,\!3\cdot\text{\sig{125}}
      \quad\Rightarrow\quad
      x\approx\text{\sig{45,73}}
    \end{equation*}
  \fi
\end{exercise}
