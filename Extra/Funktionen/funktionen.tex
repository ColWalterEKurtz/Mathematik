% #2  function (term)

\newenvironment{function}[4]
{%
  \begin{scope}[xshift=#1]
    % Funktionsgleichung
    \node[above right] at (-6, #4) {#2};
    % Koordinatenachsen
    \draw[line width=0.8pt, ->, >=latex] (-6, 0) -- (6, 0);
    \draw[line width=0.8pt, ->, >=latex] (0, #3) -- (0, #4);
    % Funktionen
    \begin{scope}
      \clip (-6, #3) rectangle (6, #4);
}%
{%
    \end{scope}   
  \end{scope}
}
