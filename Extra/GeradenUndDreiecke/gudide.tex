\paragraph{Aufgabe 1}
% -------------------

% -------------------

mit den Geraden $g_1$ und $g_2$ ein Dreieck.

  \item Bestimme den Umfang dieses Dreiecks.
  \item Bestimme den Flächeninhalt dieses Dreiecks.
  \item Bestimme die Innenwinkel dieses Dreiecks.
\end{enumerate}
