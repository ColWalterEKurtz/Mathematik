    % ----------------------------------
    \paragraph{Höhe eines Dreiecks}~\par
    % ----------------------------------
    \begin{minipage}{0.3\textwidth}
      \begin{tikzpicture}
        \coordinate (A) at (0, 0);
        \coordinate (B) at (1, 0);
        \coordinate (C) at (1, 2);
        \draw (A) -- (B) -- (C) -- cycle;
        \fill[fill=black] (A) circle[radius=1pt];
        \fill[fill=black] (B) circle[radius=1pt];
        \fill[fill=black] (C) circle[radius=1pt];
        \node[below left]  at (A) {$A$};
        \node[below right] at (B) {$D$};
        \node[above]       at (C) {$S$};
        \path (A) -- node[below]{{\small$\displaystyle\frac{a}{2}$}} (B);
        \path (A) -- node[above left]{{\small$a$}}                   (C);
        \path (B) -- node[right]{{\small$h_{d}$}}                    (C);
      \end{tikzpicture}%
    \end{minipage}%
    \rule[-4.5\baselineskip]{0pt}{9.5\baselineskip}%
    \begin{minipage}{0.6\textwidth}
      \begin{equation*}
        \begin{split}
          a^{2}&=h_{d}^{2}+\left(\frac{a}{2}\right)^{2}\\[2ex]
          h_{d}&=\sqrt{a^{2}-\left(\frac{a}{2}\right)^{2}}\\[2ex]
               &=\frac{\sqrt{3}}{2}a
        \end{split}
      \end{equation*}%
    \end{minipage}%
  \end{minipage}\hfill
  % rechts
  \begin{minipage}[t]{0.49\textwidth}

    \paragraph{Flächen-Kanten-Winkel}~\par
    % ------------------------------------

      \begin{tikzpicture}
        \coordinate (A) at (0, 0);
        \coordinate (B) at (1, 0);
        \coordinate (C) at (1, 2);
        \draw (A) -- (B) -- (C) -- cycle;
        \fill[fill=black] (A) circle[radius=1pt];
        \fill[fill=black] (B) circle[radius=1pt];
        \fill[fill=black] (C) circle[radius=1pt];
        \begin{scope}
          \clip (A) -- (B) -- (C) -- cycle;
          \draw (A) circle[radius=7mm];
          \node at ([shift={(31.7:4mm)}]A) {{\small$\alpha$}};
        \end{scope}
        \node[below left]  at (A) {$A$};
        \node[below right] at (B) {$M$};
        \node[above]       at (C) {$S$};
        \path (A) -- node[below]{{\small$\displaystyle\frac{2h_{d}}{3}$}} (B);
        \path (A) -- node[above left]{{\small$a$}}                        (C);
        \path (B) -- node[right]{{\small$h_{t}$}}                         (C);
      \end{tikzpicture}%
    \end{minipage}%
    \rule[-4.5\baselineskip]{0pt}{9.5\baselineskip}%
    \begin{minipage}{0.6\textwidth}
      \begin{equation*}
        \begin{split}
          \cos\alpha&=\frac{2h_{d}}{3a}=\frac{\sqrt{3}a}{3a}=\frac{\sqrt{3}}{3}\\[2ex]
              \alpha&=\arccos\frac{\sqrt{3}}{3}\\[2ex]
                    &\approx54,\!73561^{\circ}
        \end{split}
      \end{equation*}%
    \end{minipage}%
  \end{minipage}%
\end{minipage}
