      {Ableitungsregeln 2}
  \ifproblem\problem\par
    % <PROBLEM>
    Bestimmen Sie die Funktionsgleichung von $f'$.
    \begin{alignat*}{8}
        \text{a)}&\; & f(x)&=ax^2+bx+c
      & \quad&\quad
      & \text{b)}&\; & f(x)&=ax+c
      & \quad&\quad
      & \text{c)}&\; & f(x)&=x^{c+1}
      \\
        \text{d)}&\; & f(t)&=t^2+3t
      & \quad&\quad
      & \text{e)}&\; & f(x)&=x-t
      & \quad&\quad
      & \text{f)}&\; & f(t)&=x-t
    \end{alignat*}
    % </PROBLEM>
  \fi
  %\ifoutline\outline\par
    % <OUTLINE>
    % </OUTLINE>
  %\fi
  \ifoutcome\outcome
    % <OUTCOME>
    \begin{alignat*}{8}
        \text{a)}&\; & f'(x)&=2ax+b
      & \quad&\quad
      & \text{b)}&\; & f'(x)&=a
      & \quad&\quad
      & \text{c)}&\; & f'(x)&=(c+1)\cdot x^c
      \\
        \text{d)}&\; & f'(t)&=2t+3
      & \quad&\quad
      & \text{e)}&\; & f'(x)&=1
      & \quad&\quad
      & \text{f)}&\; & f'(t)&=-1
    \end{alignat*}
    % </OUTCOME>
  \fi
\end{exercise}
