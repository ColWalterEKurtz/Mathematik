\newlength{\dicewidth}%
\setlength{\dicewidth}{4mm}%

\newlength{\diceradius}%
\setlength{\diceradius}{1pt}%

\newcommand{\diceI}%
{%
  \draw (-0.5\dicewidth, -0.5\dicewidth) rectangle
        ( 0.5\dicewidth,  0.5\dicewidth);
  \fill (0, 0) circle[radius=\diceradius];
}%

\newcommand{\diceII}%
{%
  \draw (-0.500\dicewidth, -0.500\dicewidth) rectangle
        ( 0.500\dicewidth,  0.500\dicewidth);
  \fill (-0.165\dicewidth, -0.165\dicewidth) circle[radius=\diceradius];
  \fill ( 0.165\dicewidth,  0.165\dicewidth) circle[radius=\diceradius];
}%
