% -------
% squeeze
% -------
%
% This command squeezes itemize and enumerate environments.
%
% [#1]  leftskip
%
\newcommand{\squeeze}[1][\the\leftskip]
{%
  % shift horizontically
  \setlength{\leftskip}{#1}%
  % squeeze vertically
  \renewcommand{\itemsep}{-1ex}%
}

% -------------
% fourfoldtable
% -------------
%
% This environment typesets a fourfold table.
%
\newenvironment{fourfoldtable}
{%
  \begingroup
    % row height
    \renewcommand{\arraystretch}{1.25}%
    % orientation
    \newif\iftranspose
    \transposefalse
    \newcommand{\transpose}{\transposetrue}%
    % low-level commands
    \newcommand{\Aa}{\relax}% first row
    \newcommand{\Ab}{\relax}%
    \newcommand{\Ac}{\relax}%
    \newcommand{\Ad}{\relax}%
    \newcommand{\Ba}{\relax}% second row
    \newcommand{\Bb}{\relax}%
    \newcommand{\Bc}{\relax}%
    \newcommand{\Bd}{\relax}%
    \newcommand{\Ca}{\relax}% third row
    \newcommand{\Cb}{\relax}%
    \newcommand{\Cc}{\relax}%
    \newcommand{\Cd}{\relax}%
    \newcommand{\Da}{\relax}% fourth row
    \newcommand{\Db}{\relax}%
    \newcommand{\Dc}{\relax}%
    \newcommand{\Dd}{\relax}%
    % shorthand setter methods
    \newcommand{\setNW}[1]{\renewcommand{\Aa}{##1}}%
    \newcommand{\setNE}[1]{\renewcommand{\Ad}{##1}}%
    \newcommand{\setSW}[1]{\renewcommand{\Da}{##1}}%
    \newcommand{\setSE}[1]{\renewcommand{\Dd}{##1}}%
    \newcommand{\Apos}[1]{\renewcommand{\Ab}{##1}}%
    \newcommand{\Aneg}[1]{\renewcommand{\Ac}{##1}}%
    \newcommand{\Bpos}[1]{\renewcommand{\Ba}{##1}}%
    \newcommand{\Bneg}[1]{\renewcommand{\Ca}{##1}}%
    \newcommand{\numbers}[9]
    {%
      \renewcommand{\Bb}{\num{##1}}%
      \renewcommand{\Bc}{\num{##2}}%
      \renewcommand{\Bd}{\num{##3}}%
      \renewcommand{\Cb}{\num{##4}}%
      \renewcommand{\Cc}{\num{##5}}%
      \renewcommand{\Cd}{\num{##6}}%
      \renewcommand{\Db}{\num{##7}}%
      \renewcommand{\Dc}{\num{##8}}%
      \renewcommand{\Dd}{\num{##9}}%
    }%
    % some default values
    \Apos{\texttt{\textbackslash Apos\{\}}}%
    \Aneg{\texttt{\textbackslash Aneg\{\}}}%
    \Bpos{\texttt{\textbackslash Bpos\{\}}}%
    \Bneg{\texttt{\textbackslash Bneg\{\}}}%
    \setNE{Summe}%
    \setSW{Summe}%
    \setSE{\num{1}}%
}%
{%
    \iftranspose
      \begin{tabular}{|c||c|c||c|}
        \hline
        \Aa & \Ba & \Ca & \Da \\ \hline\hline
        \Ab & \Bb & \Cb & \Db \\ \hline
        \Ac & \Bc & \Cc & \Dc \\ \hline\hline
        \Ad & \Bd & \Cd & \Dd \\ \hline
      \end{tabular}%
    \else
      \begin{tabular}{|c||c|c||c|}
        \hline
        \Aa & \Ab & \Ac & \Ad \\ \hline\hline
        \Ba & \Bb & \Bc & \Bd \\ \hline
        \Ca & \Cb & \Cc & \Cd \\ \hline\hline
        \Da & \Db & \Dc & \Dd \\ \hline
      \end{tabular}%
    \fi
  \endgroup
}%

% -------
% fofotab
% -------
%
% This environment typesets a fourfold table.
%
% #1 [t] transpose table
%
\newenvironment{fofotab}[1][\relax]%
{%
  \begingroup
    \ifthenelse{\equal{#1}{t}}
    {%
      \newcommand{\transpose}{true}%
    }%
    {%
      \newcommand{\transpose}{false}%
    }%
    % labels
    \newcommand{\lblTval}{\relax}%
    \newcommand{\lblAval}{\texttt{\textbackslash lblA}}%
    \newcommand{\lblaval}{\texttt{\textbackslash lbla}}%
    \newcommand{\lblBval}{\texttt{\textbackslash lblB}}%
    \newcommand{\lblbval}{\texttt{\textbackslash lblb}}%
    % setting labels
    \newcommand{\lblT}[1]{\renewcommand{\lblTval}{##1}}%
    \newcommand{\lblA}[1]{\renewcommand{\lblAval}{##1}}%
    \newcommand{\lbla}[1]{\renewcommand{\lblaval}{##1}}%
    \newcommand{\lblB}[1]{\renewcommand{\lblBval}{##1}}%
    \newcommand{\lblb}[1]{\renewcommand{\lblbval}{##1}}%
    % numberic values
    \newcommand{\andABval}{\texttt{\textbackslash andAB}}%
    \newcommand{\andaBval}{\texttt{\textbackslash andaB}}%
    \newcommand{\andAbval}{\texttt{\textbackslash andAb}}%
    \newcommand{\andabval}{\texttt{\textbackslash andab}}%
    \newcommand{\sumAval}{\texttt{\textbackslash sumA}}%
    \newcommand{\sumaval}{\texttt{\textbackslash suma}}%
    \newcommand{\sumBval}{\texttt{\textbackslash sumB}}%
    \newcommand{\sumbval}{\texttt{\textbackslash sumb}}%
    \newcommand{\totalval}{\texttt{\textbackslash total}}%
    % setting numberic values
    \newcommand {\andAB} [1] {\renewcommand{\andABval}{##1}}%
    \newcommand {\andaB} [1] {\renewcommand{\andaBval}{##1}}%
    \newcommand {\andAb} [1] {\renewcommand{\andAbval}{##1}}%
    \newcommand {\andab} [1] {\renewcommand{\andabval}{##1}}%
    \newcommand {\sumA}  [1] {\renewcommand{\sumAval}{##1}}%
    \newcommand {\suma}  [1] {\renewcommand{\sumaval}{##1}}%
    \newcommand {\sumB}  [1] {\renewcommand{\sumBval}{##1}}%
    \newcommand {\sumb}  [1] {\renewcommand{\sumbval}{##1}}%
    \newcommand {\total} [1] {\renewcommand{\totalval}{##1}}%
    % vertical space
    \renewcommand{\arraystretch}{1.25}%
}%
{%
    \ifthenelse{\equal{\transpose}{true}}%
    {%
      \begin{tabular}{|c||c|c||c|}
        \hline
        \lblTval & \lblAval  & \lblaval  & Summe     \\
        \hline
        \hline
        \lblBval & \andABval & \andaBval & \sumBval  \\
        \hline
        \lblbval & \andAbval & \andabval & \sumbval  \\
        \hline
        \hline
        Summe   & \sumAval  & \sumaval  & \totalval \\
        \hline
      \end{tabular}%
    }%
    {%
      \begin{tabular}{|c||c|c||c|}
        \hline
        \lblTval & \lblBval  & \lblbval  & Summe     \\
        \hline
        \hline
        \lblAval & \andABval & \andAbval & \sumAval  \\
        \hline
        \lblaval & \andaBval & \andabval & \sumaval  \\
        \hline
        \hline
        Summe   & \sumBval  & \sumbval  & \totalval \\
        \hline
      \end{tabular}%
    }%
  \endgroup
}%
