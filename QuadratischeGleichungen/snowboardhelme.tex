\begin{exercise}

      {Snowboardhelme}
  \ifproblem\problem
    Ein Unternehmen bietet Snowboardhelme für \eur{39} das Stück an. Eine
    Marktanalyse ergab, dass sich der tägliche Gewinn $G$ (in \officialeuro)
    bei einem Verkaufspreis $x$ (in \officialeuro) mit folgender Formel
    berechnen lässt:
    \begin{equation*}
      G=-x^{2}+70x-1000
    \end{equation*}
    \begin{enumerate}[a)]
      \item Zu welchem Preis sollte das Unternehmen die Snowboardhelme
            anbieten?
      \item Wie groß wäre dann der tägliche Gewinn?
    \end{enumerate}
  \fi
  %\ifoutline\outline
  %\fi
  %\ifoutcome\outcome
  %\fi
\end{exercise}
