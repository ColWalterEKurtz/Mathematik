\begin{exercise}
      {ID-3de9e84472c58ed287ae986370783d911591a1b7}
      {Schutzwand}
  \ifproblem\problem\par
    Die Intensität einer Gamma-Strahlung wird durch eine Schutzwand
    von \sicm{10} Dicke um \pc{20} reduziert.
    \begin{enumerate}[a)]
      \item Wie groß ist die Intensität dieser Strahlung hinter
            einer \sicm{50} dicken Wand aus gleichem
            Material?
      \item Wie dick muss die Wand (aus gleichem Material) gemacht
            werden, damit dahinter nur noch \pc{10} der ursprünglichen
            Intensität vorhanden ist?
    \end{enumerate}
  \fi
  %\ifoutline\outline\par
  %\fi
  \ifoutcome\outcome\par
    \begin{enumerate}[a)]
      \item Die Intensität hinter einer \sicm{50} dicken Wand
            würde bei etwa \pc{32.768} liegen.
      \item Die Wand müsste etwa \sicm{103.189} dick sein, damit die
            Intensität der Gam\-ma-Strah\-lung auf \pc{10} absinkt.
    \end{enumerate}
  \fi
\end{exercise}
