\begin{exercise}
      {ID-b38b61eecf32865420b5a9b42b9a5feeb41919d2}
      {Wirtschaftswachstum}
  \ifproblem\problem\par
    Derzeit gibt es kein politisches System auf der Erde, das nicht
    auf Wirtschaftswachstum setzt. \pc{4} Wachstum gelten als
    wünschenswert und maßvoll: also jedes Jahr \pc{4} mehr als im
    Vergleich zum Vorjahr. Um wie viel Prozent wäre also bei diesem
    Wachstum die Wirtschaft nach 2, 10 bzw. 50 Jahren gewachsen?
  \fi
  %\ifoutline\outline\par
  %\fi
  \ifoutcome\outcome\par
    \begin{enumerate}[a)]
      \item Nach 2  Jahren wäre die Wirtschaft um etwa \pc{8.160} gewachsen.
      \item Nach 10 Jahren wäre die Wirtschaft um etwa \pc{48.024} gewachsen.
      \item Nach 50 Jahren wäre die Wirtschaft um etwa \pc{610.668} gewachsen.
    \end{enumerate}
  \fi
\end{exercise}
