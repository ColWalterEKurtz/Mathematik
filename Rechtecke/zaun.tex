\begin{exercise}
      {ID-2e77e688457c5f54cabfbd6fe149b0961f44ec3a}
      {Zaun}
  \ifproblem\problem
    Die Einzäunung einer Fläche von \simm{576} kostet \eur{80} pro Meter.
    Wie viel kostet der Zaun, wenn die Fläche quadratisch ist?
    Wie viel kostet der Zaun, wenn die Fläche rechteckig und
    eine Seite \simeter{36} lang ist?
  \fi
  %\ifoutline\outline
  %\fi
  \ifoutcome\outcome
    \begin{enumerate}[a)]
      \item Wenn die Fläche quadratisch ist,
            kostet der Zaun insgesamt \eur{7680}.
      \item Wenn eine Seite der Fläche \simeter{36} lang ist,
            kostet der Zaun insgesamt \eur{8320}.
    \end{enumerate}
  \fi
\end{exercise}
