{%
  \addvspace{\baselineskip}\par%
  \refstepcounter{myseccounter}%

% Erzeugt eine Nummer am Anfang einer Zeile
\newcommand{\formnum}
{%
  \refstepcounter{formularcounter}%
  \makebox[4em][r]{\textbf{\theformularcounter}\enskip}%
}

{%

  \ensuremath{\displaystyle#1}%
  \bigskip%
}
