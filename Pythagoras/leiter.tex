\begin{exercise}
      {ID-4a59935dd5b3d70fd9474fec3845306c9716bf24}
      {Leiter}
  \ifproblem\problem
    \begin{enumerate}[a)]
      \item Der Fuß einer Leiter steht \simeter{2.5} vor einer Hauswand. Die Leiter
            erreicht ein \simeter{6} hoch gelegenes Fenster. Wie lang ist die Leiter?
      \item Eine \simeter{12.5} lange Leiter lehnt an einer Hauswand. Das untere
            Leiterende steht dabei \simeter{3.5} von der Wand entfernt. In welcher
            Höhe liegt die Leiter an der Hauswand an?
    \end{enumerate}
  \fi
  %\ifoutline\outline
  %\fi
  \ifoutcome\outcome
    \begin{enumerate}[a)]
      \item Die Leiter ist \simeter{6.5} lang.
      \item Die Leiter liegt in einer Höhe von \simeter{12} an der Wand an.
    \end{enumerate}
  \fi
\end{exercise}
