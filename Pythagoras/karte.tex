\begin{exercise}
      {ID-bd8d8511f59aac08e41f175654b44fb36ab1b512}
      {Karte}
  \ifproblem\problem\par
    Auf einer Karte im Maßstab 1\,:\,\num{25000} wird ein rechtwinkliges Dreieck
    $ABC$ markiert. Die Länge der Hypotenuse $\overline{BC}$ beträgt \sicm{10}.
    Die Kathete $\overline{AB}$ wird mit \sicm{2.8} gemessen. Bestimme die
    tatsächliche Länge der Strecke $\overline{AC}$ in Meter.
  \fi
  %\ifoutline\outline\par
  %\fi
  \ifoutcome\outcome\par
    Die Stecke $\overline{AC}$ besitzt in Realität eine Länge von \simeter{2400}.
  \fi
\end{exercise}
