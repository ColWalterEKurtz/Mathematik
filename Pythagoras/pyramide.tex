\begin{exercise}
      {ID-4d3b12c9fa8fee2360ad69b346c8cb1e9034d98f}
      {Pyramide}
  \ifproblem\problem
    Die abgebildete Pyramide hat eine quadratische Grundfläche mit der
    Seitenlänge $a$. Die schrägen Seitenflächen haben die Seitenlänge $s$
    und die Seitenhöhe $h_{s}$. Die Höhe der Pyramide ist $h$. Alle
    Maßangaben sind in cm. Vervollständige die Tabelle.\par
    \begin{minipage}{0.35\textwidth}
      \centering
      \begin{tikzpicture}[scale=0.68]
        % Koordinaten
        \coordinate (A) at (  0.6250,   0.7500);
        \coordinate (B) at ( -0.8750,  -0.2500);
        \coordinate (C) at (  3.1250,  -0.2500);
        \coordinate (D) at (  4.6250,   0.7500);
        \coordinate (E) at (  1.8750,   4.2500);
        \coordinate (F) at (  1.8750,   0.2500);
        \coordinate (G) at (  3.8750,   0.2500);
        % Koordinaten fuer die Beschriftung
        \coordinate (HPHA)  at ($(F)!0.47!(E)$);
        \coordinate (HPHE)  at ([shift={(150:2cm)}]HPHA);
        \coordinate (HPHsA) at ($(G)!0.404!(E)$);
        \coordinate (HPHsE) at ([shift={(30:1cm)}]HPHsA);
        \coordinate (HPSA)  at ($(D)!0.6!(E)$);
        \coordinate (HPSE)  at ([shift={(30:1cm)}]HPSA);
        % unten
        \draw[dashed] (A) -- (B);
        \draw         (B) -- (C);
        \draw         (C) -- (D);
        \draw[dashed] (D) -- (A);
        % schraeg
        \draw[dashed] (A) -- (E);
        \draw         (B) -- (E);
        \draw         (C) -- (E);
        \draw         (D) -- (E);
        % Diagonalen
        \draw[line width=0.7pt, dotted] (A) -- (C);
        \draw[line width=0.7pt, dotted] (B) -- (D);
        % Hoehen
        \draw[line width=0.7pt, dotted] (E) -- (F);
        \draw[line width=0.7pt, dotted] (E) -- (G);
        % Beschriftung
        \path (B) -- node[below]       {{\small$a$}} (C);
        \path (C) -- node[below right] {{\small$a$}} (D);
        \draw[very thin] (HPHA)  -- (HPHE)  node[shift=(150:2mm)] {{\small$h$}};
        \draw[very thin] (HPHsA) -- (HPHsE) node[shift=(30:3mm)] {{\small$h_{s}$}};
        \draw[very thin] (HPSA)  -- (HPSE)  node[shift=(30:2mm)] {{\small$s$}};
      \end{tikzpicture}
    \end{minipage}%
    \hfill

        \hline

        \hline
        $h_{s}$ &   &   & 3 &   &   & 5 & 6 \\
        \hline
        $h$     & 5 & 3 &   &   & 4 &   & 5 \\
        \hline
      \end{tabular}
    \end{minipage}
