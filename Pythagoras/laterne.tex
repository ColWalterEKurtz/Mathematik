\begin{exercise}
      {ID-3e47be24074bec6a8ab79ee6aa2212b9442f6204}
      {Laterne}
  \ifproblem\problem\par
    Eine Lampe hängt zwischen zwei \simeter{5} hohen und \simeter{12} voneinander
    entfernten Laternenmasten. Wie hoch in darf ein darunter fahrendes Fahrzeug
    maximal sein, wenn die Lampe an einem \simeter{12.1} langen Stahlseil
    befestigt ist?
  \fi
  %\ifoutline\outline\par
  %\fi
  \ifoutcome\outcome\par
    Ein Fahrzeug darf höchstens \simeter{4.22} hoch sein,
    um unter der Lampe hindurch fahren zu können.
  \fi
\end{exercise}
